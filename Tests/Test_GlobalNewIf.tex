\chapter{\Cmd{GlobalNewIf} Tests}

% ================================================================ %
%                       Definition Checks                          %
% ================================================================ %
\section{Definition Checks}

% ------------------------------------------------------- %
%         Define TestIfa through \GlobalNewIf             %
% ------------------------------------------------------- %
\GlobalNewIf{TestIfa}

Test if \Cmd{GlobalNewIf} can successfully define the if-statement:\\
% ----------------------------------- 
% if-statement check
\ifx \ifTestIfa \UnDefined
    \rule{2em}{0pt}\Cmd{ifTestIfa} was unsuccessfully defined by \Cmd{GlobalNewIf}.
\else
    \rule{2em}{0pt}\Cmd{ifTestIfa} was successfully defined by \Cmd{GlobalNewIf}.
\fi\\
%
% ----------------------------------- 
% true command check
\ifx \TestIfatrue \UnDefined
    \rule{2em}{0pt}\Cmd{TestIfatrue} was unsuccessfully defined by \Cmd{GlobalNewIf}.
\else
    \rule{2em}{0pt}\Cmd{TestIfatrue} was successfully defined by \Cmd{GlobalNewIf}.
\fi\\
%
% ----------------------------------- 
% false command check
\ifx \TestIfafalse \UnDefined
    \rule{2em}{0pt}\Cmd{TestIfafalse} was unsuccessfully defined by \Cmd{GlobalNewIf}.
\else
    \rule{2em}{0pt}\Cmd{TestIfafalse} was successfully defined by \Cmd{GlobalNewIf}.
\fi


% ------------------------------------------------------- %
%       Define TestIfb through \newif TeX primitive       %
% ------------------------------------------------------- %
\newif\ifTestIfb
\Cmd{ifTestIfb} is defined through \Cmd{newif} for a scoping comparison.





% ================================================================ %
%                          Scope Checks                            %
% ================================================================ %
\section{Scope Checks}
\TestIfatrue
\TestIfbtrue

\ifTestIfa
    \Cmd{ifTestIfa} is set to true in the main block.
\else
    \Cmd{ifTestIfa} is set to false in the main block.
\fi\\
\ifTestIfb
    \Cmd{ifTestIfb} is set to true in the main block.
\else
    \Cmd{ifTestIfb} is set to false in the main block.
\fi

\begin{minipage}{5in}
    \rule{3in}{0.1em}\\
    \TestIfafalse
    \TestIfbfalse
    \ifTestIfa
        \Cmd{ifTestIfa} is set to true in this minipage.
    \else
        \Cmd{ifTestIfa} is set to false in this minipage.
    \fi\\
    \ifTestIfb
        \Cmd{ifTestIfb} is set to true in this minipage.
    \else
        \Cmd{ifTestIfb} is set to false in this minipage.
    \fi\\
    \rule{3in}{0.1em}
\end{minipage}

    \ifTestIfa
        \Cmd{ifTestIfa} has reverted back to true after exiting the minipage.
    \else
        \Cmd{ifTestIfa} is still false after exiting the minipage.
    \fi\\
    \ifTestIfb
        \Cmd{ifTestIfb} has reverted back to true after exiting the minipage.
    \else
        \Cmd{ifTestIfb} is still false after exiting the minipage.
    \fi
